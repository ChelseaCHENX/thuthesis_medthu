% 中英文摘要和关键字
\renewcommand{\chaptermark}[1]{\markboth{#1}{}} 
\begin{abstract}
  \markboth{摘要}{}
  肿瘤异质性
  % 关键词用“英文逗号”分隔，输出时会自动处理为正确的分隔符
  \thusetup{
    keywords = {异质性, 精准医疗},
  }
\end{abstract}

\renewcommand{\chaptermark}[1]{\markboth{#1}{}} 
\begin{abstract*}
  \markboth{Abstract}{}
  Tumor heterogeneity
  This paper aims for 
  % Use comma as separator when inputting
  \thusetup{
    keywords* = {heterogeneity, precision medicine},
  }
\end{abstract*}
